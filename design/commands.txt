USERS
    CREATE TABLE USERS(UID INT PRIMARY KEY AUTO_INCREMENT, EMAIL VARCHAR(255) NOT NULL, NAME VARCHAR(255) NOT NULL, USERNAME VARCHAR(255) NOT NULL, PASSWORD VARCHAR(255) NOT NULL, JOINED TIMESTAMP DEFAULT CURRENT_TIMESTAMP());

DATASETS
    CREATE TABLE DATASETS(DID INT PRIMARY KEY AUTO_INCREMENT, NAME VARCHAR(255) NOT NULL, DESCRIPTION VARCHAR(5000) NOT NULL, CREATOR INT NOT NULL, FORMAT VARCHAR(10) NOT NULL, SIZE FLOAT NOT NULL, PUBLIC BOOLEAN NOT NULL, CONTRIBUTABLE BOOLEAN NOT NULL, FILENAME VARCHAR(255) NOT NULL, LAST_MODIFIED DATETIME DEFAULT CURRENT_TIMESTAMP(), CREATED DATETIME DEFAULT CURRENT_TIMESTAMP(), DOWNLOADS INT DEFAULT 0, CUSTOM_FUNCTION INT DEFAULT -1, FOREIGN KEY(CREATOR) REFERENCES USERS(UID) ON DELETE CASCADE);
    CREATE TRIGGER CONTRIBUTION_TRIGGER AFTER INSERT ON DATASETS 
    FOR EACH ROW
        INSERT INTO CONTRIBUTIONS(DID, UID, MESSAGE) VALUES(NEW.DID, NEW.CREATOR, 'File Upload')
CONTRIBUIONS
    CREATE TABLE CONTRIBUTIONS(CID INT PRIMARY KEY AUTO_INCREMENT, DID INT NOT NULL, UID INT NOT NULL, MESSAGE VARCHAR(5000) NOT NULL, MID INT, FOREIGN KEY(UID) REFERENCES USERS(UID) ON DELETE CASCADE, FOREIGN KEY(DID) REFERENCES DATASETS(DID) ON DELETE CASCADE, FOREIGN KEY(MID) REFERENCES MODELS(MID) ON DELETE CASCADE)

MESSAGES
    CREATE TABLE MESSAGES(MID INT PRIMARY KEY AUTO_INCREMENT, UID INT NOT NULL UNIQUE, SUBJECT VARCHAR(1000) NOT NULL, MESSAGE VARCHAR(5000) NOT NULL, FOREIGN KEY(UID) REFERENCES USERS(UID) ON DELETE CASCADE)

REQUESTS
    CREATE TABLE REQUESTS(RID INT PRIMARY KEY AUTO_INCREMENT, DID INT NOT NULL, UID INT NOT NULL, MESSAGE VARCHAR(5000) NOT NULL, TYPE VARCHAR(15) NOT NULL, STATUS INT DEFAULT 0, FOREIGN KEY(UID) REFERENCES USERS(UID) ON DELETE CASCADE, FOREIGN KEY(DID) REFERENCES DATASETS(DID) ON DELETE CASCADE )

FEEDBACK
    CREATE TABLE FEEDBACK(USERNAME VARCHAR(255) NOT NULL, EMAIL VARCHAR(255) NOT NULL, MESSAGE VARCHAR(5000) NOT NULL)

MODELS
    CREATE TABLE MODELS(MID INT PRIMARY KEY AUTO_INCREMENT, DID INT, UID INT, FILENAME VARCHAR(255), DESCRIPTION VARCHAR(5000), FOREIGN KEY(DID) REFERENCES DATASETS(DID) ON DELETE CASCADE, FOREIGN KEY(UID) REFERENCES USERS(UID) ON DELETE CASCADE)

MODEL REQUESTS
    CREATE TABLE MODEL_REQUESTS(MRID INT PRIMARY KEY AUTO_INCREMENT, DID INT, UID INT, MESSAGE VARCHAR(5000), FILENAME VARCHAR(255), TYPE VARCHAR(15) NOT NULL, STATUS INT DEFAULT 0, FOREIGN KEY(DID) REFERENCES DATASETS(DID) ON DELETE CASCADE, FOREIGN KEY(UID) REFERENCES USERS(UID) ON DELETE CASCADE)
